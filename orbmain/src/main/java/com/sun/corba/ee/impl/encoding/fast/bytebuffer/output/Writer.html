036 package com.sun.corba.ee.impl.encoding.fast.bytebuffer ;

038 import java.io.Closeable ;
039 
040 import java.nio.ByteOrder ;
041 
042 /** Write the primitive Java types and arrays of primitives to some data sink.  
043  * This may include internal buffering for efficiency reasons.
044  */
045 public interface Writer extends Closeable {
046     /** Interface that defines how full buffers are handled.  Typically this
047      * implies some kind of IO, and eventual disposition of the used buffer.
048      */
049     public interface BufferHandler {
050         /** Dispose of the current ByteBuffer and return a new ByteBuffer
051          * which will be written to from position to limit.
052          * Note that current may be null.
053          */
054         BufferWrapper overflow( BufferWrapper current ) ;
055 
056         /** Dispose of the current ByteBuffer.
057          * Used when closing a ByteBufferWriter.
058          */
059         void close( BufferWrapper current ) ;
060     }
061 
062     /** Get the BufferHandler in use for this stream.
063      */
064     BufferHandler bufferHandler() ;
065 
066     /** Return the ByteOrder of the stream.
067      * All streams default to big endian byte order.
068      */
069     ByteOrder order() ;
070 
071     /** Set the ByteOrder of the stream.
072      */
073     void order( ByteOrder byteOrder ) ;
074 
075     /** Make sure that all data that has been written is
076      * flushed from the stream to its destination.
077      */
078     void flush() ;
079 
080     void putBoolean( boolean data ) ;
081 
082     void putByte( byte data ) ;
083 
084     void putChar( char data ) ;
085 
086     void putShort( short data ) ;
087 
088     void putInt( int data ) ;
089 
090     void putLong( long data ) ;
091 
092     void putFloat( float data ) ;
093 
094     void putDouble( double data ) ;
095 
096     void putBooleanArray( boolean[] data ) ;
097 
098     void putByteArray( byte[] data ) ;
099 
100     void putCharArray( final char[] data ) ;
101 
102     void putShortArray( short[] data ) ;
103 
104     void putIntArray( int[] data ) ;
105 
106     void putLongArray( long[] data ) ;
107 
108     void putFloatArray( float[] data ) ;
109 
110     void putDoubleArray( double[] data ) ;
111 }
