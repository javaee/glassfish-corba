036 package com.sun.corba.se.impl.encoding.fast.bytebuffer ;
037 
038 import java.nio.ByteBuffer ;
039 
040 /** Base class for implementing an Allocator.
041  * All subclasses must implement dispose, releaseSlab and obtainSlab, as well as the undefined
042  * methods in the Allocator interface: maxAllocationSize, bufferType, and close.
043  * This class handles allocation of BufferWrappers from the current Slab.
044  */
045 public abstract class AllocatorBase implements Allocator {
046     private int headerSize ;
047     private Slab currentSlab ;
048     private boolean closed = false ;
049 
050     protected final void checkClosed() {
051         if (closed) {
052             throw new IllegalStateException( "Allocator is closed" ) ;
053         }
054     }
055 
056     AllocatorBase( int headerSize ) {
057         this.headerSize = headerSize ;
058 
059         currentSlab = obtainSlab() ;
060     }
061 
062     public final int headerSize() {
063         checkClosed() ;
064         return headerSize ;
065     }
066 
067     // Used only by BufferWrapperImpl
068     abstract void dispose( Slab slab, ByteBuffer store ) ;
069 
070     abstract void releaseSlab( Slab slab ) ;
071 
072     abstract Slab obtainSlab() ; 
073 
074     public final BufferWrapper allocate( int size ) {
075         checkClosed() ;
076 
077         if (size > maxAllocationSize()) {
078             throw new IllegalArgumentException( "Request size " + size 
079                 + " is larger than maximum allocation size " + maxAllocationSize() ) ;
080         }
081 
082         if (currentSlab.sizeAvailable() < (size + headerSize)) {
083             currentSlab.markFull() ;
084             releaseSlab( currentSlab ) ;
085             currentSlab = obtainSlab() ;
086         }
087 
088         BufferWrapper result = new BufferWrapper( this, currentSlab, headerSize, size ) ;
089         return result ;
090     }
091 
092     public final BufferWrapper allocate( int minSize, int maxSize ) {
093         checkClosed() ;
094 
095         if (minSize > maxAllocationSize()) {
096             throw new IllegalArgumentException( "Minimum request size " + minSize 
097                 + " is larger than maximum allocation size " + maxAllocationSize() ) ;
098         }
099 
100         int remaining = currentSlab.sizeAvailable() - headerSize ; 
101         int allocSize = maxSize ;
102         if ((remaining >= minSize) && (remaining <= maxSize)) {
103             allocSize = remaining ;
104         }
105 
106         return allocate( allocSize ) ;
107     }
108 
109     public final void close() {
110         checkClosed() ;
111         releaseSlab( currentSlab ) ;
112         closed = true ;
113     }
114 }
