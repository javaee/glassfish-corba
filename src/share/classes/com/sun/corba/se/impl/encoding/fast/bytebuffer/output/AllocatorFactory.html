036 package com.sun.corba.se.impl.encoding.fast.bytebuffer ;
037 
038 import java.io.Closeable ;
039 
040 /** Factory for Allocators (pooled and non-pooled) and SlabPools.
041  */
042 public class AllocatorFactory {
043     /** makeAllocator and makePoolAllocator must always have headerSize at least MIN_HEADER_SIZE.
044      */
045     int MIN_HEADER_SIZE = 8 ;
046 
047     private AllocatorFactory() {}
048 
049     /** Return an Allocator that allocates from one slab at a time, creating a new slab
050      * as needed.  Space is recovered by the garbage collector.
051      * This version will create a new slab allocator when available space is exhausted.
052      * It will only return null if an attempt is made to allocate a BufferWrapper larger
053      * than allocatorSize - headerSize bytes.
054      * trim is supported, but only for the last BufferWrapper returned from an allocate call.
055      * @param headerSize The size of the reserved header space in an allocated BufferWraper.
056      * @param allocatorSize The total size available for all allocate calls in a single Allocator.
057      * @param bt The buffer type of the slabs created for this allocator.
058      */
059     public static Allocator makeAllocator( int headerSize, int allocatorSize, 
060         Allocator.BufferType bt ) {
061         return new AllocatorImpl( headerSize, allocatorSize, bt ) ;
062     }
063 
064     /** Create an Allocator that allocates from a pool of slabs.  All space is managed by
065      * the Slabs and SlabPool, and Slabs are recycled once they have been completely freed.
066      * @param headerSize The size of the reserved header space in an allocated BufferWraper.
067      * @param pool The SlabPool to use for Slabs for this Allocator.
068      */
069     public static Allocator makePoolAllocator( int headerSize, SlabPool pool ) {
070         return new AllocatorPoolImpl( headerSize, pool ) ;
071     }
072 
073     /** Obtain useful statistics about the SlabPool.
074      */
075     public interface SlabPool extends Closeable {
076         /** The maxAllocationSize of every Slab in this pool.
077          */
078         int maxAllocationSize() ;
079 
080         /** The BufferType of every Slab in this pool.
081          */
082         Allocator.BufferType bufferType() ;
083 
084         /** Minimum size of pool, as specified when the SlabPool was created.  
085          */
086         long minSize() ;
087 
088         /** Maximum size to which pool may expand as specified when the SlabPool was created.  
089          */
090         long maxSize() ;
091 
092         /** Current free space in pool.
093          */
094         long freeSpace() ;
095 
096         /** Total bytes disposed but not yet available for use.  This is always less
097          * than allocatedSpace.
098          */
099         long inUseSpace() ;
100 
101         /** Total bytes allocated but not yet disposed.
102          */
103         long allocatedSpace() ;
104     }
105 
106     /** Create a SlabPool from which Allocators may be created.
107      * @param maxAllocationSize The maximum size that can be satisfied by a call to Allocator.allocate().
108      * @param minSize The minimum size of the pool in bytes.
109      * @param maxSize The maximum size to which a pool may grow.
110      * @param by The buffer type used in this pool.
111      */
112     public SlabPool makeSlabPool( int maxAllocationSize, long minSize, long maxSize, 
113         Allocator.BufferType bt ) {
114 
115         return new SlabPoolImpl( maxAllocationSize, minSize, maxSize, bt ) ;
116     }
117 }
