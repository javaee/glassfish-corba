36 package com.sun.corba.ee.impl.encoding.fast.bytebuffer ;
37 
38 import java.nio.ByteBuffer ;
39 
40 /** A simple non-pooled implementation of an Allocator.  It creates a Slab whenever the current Slab is
41  * exhausted, and relies on the garbage collector to recover the space.  Generally this should only
42  * be used with BufferType HEAP.
43  */
44 public class AllocatorImpl extends AllocatorBase {
45     private int maxAllocationSize ;
46     private Allocator.BufferType bt ;
47 
48     public AllocatorImpl( int headerSize, int maxAllocationSize, 
49         Allocator.BufferType bt ) {
50         super( headerSize ) ;
51 
52         this.maxAllocationSize = maxAllocationSize ;
53         this.bt = bt ;
54     }
55 
56     public int maxAllocationSize() {
57         checkClosed() ;
58         return maxAllocationSize ;
59     }
60 
61     public Allocator.BufferType bufferType() {
62         checkClosed() ;
63         return bt ;
64     }
65 
66     @Override
67     void dispose( Slab slab, ByteBuffer store ) {
68         // NO-OP in this version
69     }
70 
71     @Override
72     Slab obtainSlab() {
73         return new Slab( maxAllocationSize + headerSize(), bt ) ;
74     }
75 
76     @Override
77     void releaseSlab( Slab slab ) {
78         // NO-OP in this version
79     }
80 }
