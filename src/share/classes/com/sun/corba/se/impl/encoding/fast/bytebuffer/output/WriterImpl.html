048 /** Write the primitive Java type to the current ByteBuffer.  If it doesn't
049  * fit, call the BufferHandler, and write to the result, which becomes the
050  * new current ByteBuffer.  Arrays will be written across multiple ByteBuffers
051  * if necessary, but all primitives will be written to a single ByteBuffer.
052  */
053 public class WriterImpl implements Writer {
054     private BufferHandler handler ;
055     private volatile BufferWrapper buffer ;
056 
057     private boolean isClosed = false ;
058     private ByteOrder byteOrder ;
059 
060     /** Create a new ByteBufferWriter.  An instance maintains a current buffer
061      * for use in writing.  Whenever the current buffer is insufficient to hold
062      * the required data, the BufferHandler is called, and the result of the
063      * handler is the new current buffer. The handler is responsible for 
064      * the disposition of the contents of the old buffer.
065      */
066     public WriterImpl( BufferHandler handler ) {
067         this.handler = handler ;
068         byteOrder = ByteOrder.BIG_ENDIAN ;
069     }
070 
071     public BufferHandler bufferHandler() {
072         return handler ;
073     }

075     public ByteOrder order() {
076         return byteOrder ;
077     }

079     public void order( ByteOrder byteOrder ) {

081     }

083     private void overflow() {
084         // Why was this here: buffer.buffer().limit( buffer.buffer().position() ) ;
085         initBuffer( handler.overflow( buffer ) ) ;
086     }
087 
088     private void initBuffer( BufferWrapper buffer ) {
089         this.buffer = buffer ;
090         buffer.buffer().clear() ;
091         buffer.buffer().order( byteOrder ) ;
092     }
093 
094     /** Cause the overflow handler to be called even if buffer is not full.
095      */
096     public synchronized void flush() {
097         overflow() ;
098     }
099 
100     public synchronized void close() {
101         handler.close( buffer ) ;
102         buffer = null ;
103         isClosed = true ;
104     }

106     /** Ensure that the requested amount of space is available
107      */
108     public synchronized void ensure( int size ) {
109         if (isClosed)
110             throw new IllegalStateException( 
111                 "ByteBufferWriter is closed" ) ;
112 
113         if ((buffer == null) || (buffer.remaining() < size)) {
114             overflow() ;
115         }
116 
117         if (buffer.remaining() < size) {
118             throw new RuntimeException( "Newly allocated buffer is too small" ) ;
119         }
120     }

123         ensure( 1 ) ;
124         byte value = data ? (byte)1 : (byte)0 ;
125         buffer.buffer().put( value ) ;
126     }

163     public void putBooleanArray( boolean[] data ) {
164         int ctr = 0 ;
165         while (ctr < data.length) {
166             int dataSizeToWrite = Math.min( data.length - ctr, 
167                 buffer.buffer().remaining()) ;
168 
169             for (int ctr2 = ctr; ctr2<ctr+dataSizeToWrite; ctr2++) {
170                 buffer.buffer().put( (byte)(data[ctr2] ? 1 : 0) ) ;
171             }
172             ctr += dataSizeToWrite ;
173 
174             if (ctr == data.length)
175                 break ;
176 
177             overflow() ;
178         }
179     }

181     public void putByteArray( byte[] data ) {
182         int ctr = 0 ;
183         while (true) {
184             int dataSizeToWrite = Math.min( data.length - ctr, 
185                 buffer.buffer().remaining()) ;
186 
187             buffer.buffer().put( data, ctr, dataSizeToWrite ) ;
188             ctr += dataSizeToWrite ;
189             
190             if (ctr == data.length)
191                 break ;
192 
193             overflow() ;
194         }
195     }
